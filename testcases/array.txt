

int main() {
    int a[20]; 
    int b, c; 
    b = 3;
    c = 5;
    a[0] = b + c; 
    cout << a[0] << endl; // should print out 8

    a[1] = 100; 
    cout << a[1] << endl; // should print out 100

    a[2] = 200; 
    cout << a[2] << endl; // should print out 200

    a[3] = a[0] * (a[1] + c); 
    cout << a[3] << endl; // should print out 840; since 840 = 8 * (100 +5)

    return 0;
}
